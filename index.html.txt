<script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"></script>
<script src="https://unpkg.com/leaflet-routing-machine@latest/dist/leaflet-routing-machine.js"></script>
<script src="https://cdn.jsdelivr.net/npm/tesseract.js@5/dist/tesseract.min.js"></script>

<script>
    // 1. Khởi tạo Bản đồ (Dùng Esri để tránh lỗi 403)
    let map = L.map('map', { zoomControl: false }).setView([10.762622, 106.660172], 13);
    L.tileLayer('https://server.arcgisonline.com/ArcGIS/rest/services/World_Street_Map/MapServer/tile/{z}/{y}/{x}').addTo(map);

    let customers = JSON.parse(localStorage.getItem('shipper_db_final')) || [];
    let deliveryQueue = [];
    let userLocation = null;
    let routingControl = null;
    let tempMarker = null;
    let userMarker = null;

    // 2. Xử lý GPS
    navigator.geolocation.watchPosition(p => {
        userLocation = L.latLng(p.coords.latitude, p.coords.longitude);
        if (userMarker) userMarker.setLatLng(userLocation);
        else userMarker = L.circleMarker(userLocation, { radius: 9, fillColor: "#007bff", color: "#fff", weight: 3, fillOpacity: 1 }).addTo(map);
    }, null, { enableHighAccuracy: true });

    function timToiNgay() {
        if (userLocation) map.flyTo(userLocation, 16);
        else alert("Đang tìm tín hiệu GPS...");
    }

    // 3. Chọn điểm trên bản đồ
    map.on('click', e => {
        document.getElementById('custLatLng').value = `${e.latlng.lat.toFixed(6)}, ${e.latlng.lng.toFixed(6)}`;
        if (tempMarker) map.removeLayer(tempMarker);
        tempMarker = L.marker(e.latlng, { draggable: true }).addTo(map);
    });

    // 4. AI Quét ảnh (OCR)
    document.getElementById('imageInput').addEventListener('change', function(e) {
        const file = e.target.files[0];
        if (!file) return;
        const status = document.getElementById('ocrStatus');
        status.style.display = 'block';
        
        Tesseract.recognize(file, 'vie').then(({ data: { text } }) => {
            status.style.display = 'none';
            // Tìm SĐT
            const phoneMatch = text.match(/(0[3|5|7|8|9][0-9]{8})\b/);
            if (phoneMatch) document.getElementById('custPhone').value = phoneMatch[0];
            // Lấy dòng đầu làm tên khách
            const lines = text.split('\n').filter(l => l.trim().length > 3);
            if (lines.length > 0) document.getElementById('custName').value = lines[0].substring(0, 20);
            alert("Đã quét xong thông tin đơn hàng!");
        });
    });

    // 5. Quản lý khách hàng
    function luuKhachHang() {
        const name = document.getElementById('custName').value;
        const latlng = document.getElementById('custLatLng').value;
        if (!name || !latlng) return alert("Vui lòng nhập tên và chạm vào bản đồ để chọn vị trí!");

        const [lat, lng] = latlng.split(',').map(Number);
        customers.push({
            name, lat, lng,
            phone: document.getElementById('custPhone').value,
            address: document.getElementById('custAddress').value,
            note: document.getElementById('custNote').value,
            history: []
        });
        
        localStorage.setItem('shipper_db_final', JSON.stringify(customers));
        if (tempMarker) { map.removeLayer(tempMarker); tempMarker = null; }
        hienThiDanhSach();
        ['custName','custPhone','custAddress','custNote','custLatLng'].forEach(id => document.getElementById(id).value = '');
    }

    function hienThiDanhSach() {
        const container = document.getElementById('customerList');
        container.innerHTML = '<strong>📇 Danh bạ khách hàng</strong><br><br>';
        customers.forEach((c, i) => {
            const h = c.history.length ? `💡 Thói quen: ${c.history[c.history.length-1]}` : "";
            container.innerHTML += `
                <div class="customer-item">
                    <div style="display:flex; justify-content:space-between; align-items:center">
                        <strong style="font-size:16px">${c.name}</strong>
                        <a href="tel:${c.phone}" class="cust-phone">📞 Gọi</a>
                    </div>
                    <span class="cust-address">${c.address || "Vị trí đã ghim"}</span>
                    ${c.note ? `<span class="cust-note">📝 ${c.note}</span>` : ''}
                    <span class="habit-tag">${h}</span>
                    <div style="display:flex; gap:8px; margin-top:12px">
                        <button style="background:var(--primary); padding:8px; font-size:12px" onclick="themVaoQueue(${i})">+ Lộ trình</button>
                        <button style="background:var(--danger); padding:8px; font-size:12px" onclick="xoaKhach(${i})">Xong/Xóa</button>
                    </div>
                </div>`;
        });
    }

    // 6. Lộ trình & Tối ưu
    function themVaoQueue(i) {
        if (!deliveryQueue.find(q => q.lat === customers[i].lat)) {
            deliveryQueue.push(customers[i]);
            capNhatQueueUI();
        }
    }

    function capNhatQueueUI() {
        const q = document.getElementById('queueList');
        q.innerHTML = deliveryQueue.length ? deliveryQueue.map((c, i) => `
            <div class="queue-item">
                <span><b>${i+1}.</b> ${c.name}</span>
                <span onclick="deliveryQueue.splice(${i},1);capNhatQueueUI()" style="color:var(--danger); cursor:pointer; font-weight:bold">✖</span>
            </div>`).join('') : '<small style="color:#999">Chưa chọn đơn hàng...</small>';
    }

    function toiUuLoTrinh() {
        if (!userLocation || deliveryQueue.length < 2) return alert("Cần bật GPS và chọn ít nhất 2 đơn!");
        let optimized = [];
        let unvisited = [...deliveryQueue];
        let current = userLocation;
        while (unvisited.length > 0) {
            unvisited.sort((a, b) => L.latLng(current).distanceTo(L.latLng(a.lat, a.lng)) - L.latLng(current).distanceTo(L.latLng(b.lat, b.lng)));
            let next = unvisited.shift();
            optimized.push(next);
            current = L.latLng(next.lat, next.lng);
        }
        deliveryQueue = optimized;
        capNhatQueueUI();
        batDauLoTrinh();
    }

    function batDauLoTrinh() {
        if (routingControl) map.removeControl(routingControl);
        if (!userLocation || deliveryQueue.length === 0) return alert("Vui lòng chọn đơn hàng!");
        const waypoints = [userLocation, ...deliveryQueue.map(c => L.latLng(c.lat, c.lng))];
        routingControl = L.Routing.control({
            waypoints,
            language: 'vi',
            createMarker: (i, wp) => L.marker(wp.latLng).bindPopup(i === 0 ? "Bạn" : deliveryQueue[i-1].name)
        }).addTo(map);
    }

    function xoaKhach(i) {
        if (confirm("Giao xong đơn này?")) {
            const now = new Date();
            customers[i].history.push(`${now.getHours()}:${now.getMinutes()}`);
            customers.splice(i, 1);
            localStorage.setItem('shipper_db_final', JSON.stringify(customers));
            hienThiDanhSach();
            if (routingControl) map.removeControl(routingControl);
        }
    }

    hienThiDanhSach();
</script>
